## let

##### 1.块级作用域：

```js
{
  var a=1
let b=1
}
console.log(a)// 1    用var可以拿到 
console.log(b)// b is not defined
```



##### 2.不存在变量声明提前

```js
// console.log(a)
// var a = 1 //undefined

console.log(b)
let b = 1 //Cannot access 'b' before initialization
```



##### 3.暂时性死区

```js
let tmp = 123

if (true) {
    tmp = 'abc' //ReferenceError
    console.log(tmp)
    let tmp
}
```





##### 4.不允许重复声明

```js
let a = 1
let a = 10
console.log(a) //Identifier 'a' has already been declared
```





## const

声明常量的

```js
//const的本质并不是变量的值不动，而是变量指向的那个内存地址所保存的数据不得改动。对于简单类型的数据，值就保存在变量指向的那个内存地址，因此等同于常量，但对于复合型的数据，变量指向的内存地址只是一个指向实际数据的指针。

```



##### 1.常量的值不能重新赋值

```js
const a = 1
a = 2 //Assignment to constant variable.
```



##### 2.对于数组和对象的元素修改，不算做对常量的修改，不会报错（内存地址没有变）

```js
const obj = {}

obj.a = 1
console.log(obj) //{ a: 1 }
```

##### 解构函数的默认值

ES6内部使用严格相等运算符（===），判断一个位置是否有值，所以，只有一个数组成员严格等于undefined，默认值才会生效。

```js
let [foo = true] = []
console.log('foo===>', foo) //foo===> true

let [foo = true] = [1]
console.log('foo===>', foo) //foo===> 1

let [foo = 2] = [10]
console.log('foo===>', foo) //foo===> 10

let [x, y = 'b'] = ['a', undefined]
//x='a'  y='b'
console.log(y) //b

let [x = 1] = [null]
console.log(x)//null
```



##### 对象的解构

```js
let { foo, bar } = { foo: 'aaa', bar: 'bbb' }
console.log(foo) //a

let { foo1, bar } = { foo: 'aaa', bar: 'bbb' }
console.log(foo1)//undefined

let { foo } = { bar: 'baz' }
console.log(foo)//undefined

//改名
let obj = { foo: 'aaa', bar: 'bbb' }
let { foo: baz } = obj//等同于let baz=obj.foo
console.log(baz) //aaa

//默认值
let obj = {}
console.log(obj.x) //undefined
let { x = 3 } = obj
//等同于let { x = 3 } = {}
console.log(x) //3

let obj = { x: 1 }
let { x, y = 5 } = obj
//等同于let { x, y = 5 } = {x:1}
console.log(x) //1
console.log(y) //5
```

## 箭头函数

##### 1.箭头函数的简写

```js
//形参有且只有一个时
let func = function(x) {
    return x * x
}
console.log(func(2))
let func = (x) => {
    return x * x
}
console.log(func(2))
//代码只有一条语句时
let func = x => x * x
console.log(func(2))

//形参有两个时
let func = (x, y) => x * y
console.log(func(2,3))


```

##### 2.箭头函数的this

箭头函数的this是静态的，this始终指向函数声明时所在作用域下的this值。

```js
var obj = {
    birth: 1998,
    getAge: function() {
        var b = this.birth
        var fn = () => {

            let fn1 = () => {
                console.log(new Date().getFullYear()) //2020
                console.log(this.birth) //1998
                return new Date().getFullYear() - this.birth;
            }

            return fn1()

        }
        return fn()
    }

}
console.log(obj.getAge()) //22
```



不是箭头函数时

```js
//global.birth = 1998
var obj = {
    birth: 1998,
```

## Map/Set

#### Set，类似于数组，但是成员的值都是唯一的不能重复

用处：去重

```js
let s = new Set()
// console.log(s) //Set(0) {}
let array = [2, 3, 5, 4, 5, 3, 2]
array.forEach(function(item) {
    s.add(item)
})
console.log(s) //Set(4) { 2, 3, 5, 4 }
console.log(s.has(2)) //true
console.log(s.size) //4
s.delete(2)
console.log(s)//Set(3) { 3, 5, 4 }

```

#### Map

```js
let map = new Map()

let obj = {
    a: 1
}
console.log(obj)//{ a: 1 }

map.set(obj, '1234')
console.log(map) //Map(1) { { a: 1 } => '1234' }

console.log(map.get(obj)) //1234

//map.delete(obj)
//console.log(map) //Map(0) {}

let a = 1
map.set(a, '1111')
console.log(map) //Map(2) { { a: 1 } => '1234', 1 => '1111' }

```



#### map用处:（计数，计算这个数字出现了几次）

```js
let map = new Map()
let array = [2, 3, 5, 4, 5, 3, 2, 2, 2, 2, 2]
array.forEach((item) => {
    //如果map里面没有这个数字进if
    if (!map.get(item)) {
      //新插入一个key-value值，代表第一次遇到这个数字
        map.set(item, 1)
    } else {
        //如果数字已经存在，获取之前计算的次数，然后+1
        map.set(item, Number(map.get(item)) + 1)
    }
})
console.log(map) //Map(4) { 2 => 6, 3 => 2, 5 => 2, 4 => 1 }
```
