   ```js
    async function async() {
      console.log('1');

      setTimeout(function () {
           resolve(888);
           console.log('2');
      		},1000)
      });

    }
    async()

   ```

   ```js
   async function async1() {
       console.log(1);
       await async2();
       console.log(3);
   }
   async function async2() {
       console.log(2);
   }
   async1();
   new Promise(function (resolved) {
       console.log(4);
   })

   ```

   ```js
   async function async1() {
       console.log(1);
       new Promise((resolved)=>{
            async2()
            resolved();
       }).then(()=>{
           console.log(3);
       })
   }
   async function async2() {
       console.log(2);
   }
   
   async1();
   
   //1,2,3
   ```
